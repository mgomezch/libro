\begin{center}
{\large DEDICATORIA}
\par\end{center}{\large \par}

\vspace*{4cm}


\noindent \begin{flushright}
\textit{A mis padres,}\\
\textit{sin su esfuerzo y apoyo incondicional}\\
\textit{nunca habría alcanzado mis metas,}\\
\textit{son el tesoro más grande que tengo.}\\
\textit{A mis hermanas,}\\
\textit{nada reemplaza su amistad y cariño.}\\

\par\end{flushright}
